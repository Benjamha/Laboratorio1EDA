import java.util.Scanner;

public class BigVigenere {
    private int[] key;
    private final String chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789ñÑ";
    private char[][] alphabet;
    private final Scanner scanner = new Scanner(System.in);
    // Constructores
    public BigVigenere() {
        System.out.print("Ingresa la key: ");
        String inputKey = scanner.nextLine();
        setKey(inputKey);
        buildAlphabet();
    }
    public BigVigenere(String numericKey) {
        setKey(numericKey);
        buildAlphabet();
    }
    private void setKey(String numericKey) {
        key = new int[numericKey.length()];
        for (int i = 0; i < numericKey.length(); i++) {
            char c = numericKey.charAt(i);
            key[i] = chars.indexOf(c);
        }
    }
    private void buildAlphabet() {
        alphabet = new char[64][64];
        for (int row = 0; row < 64; row++) {
            for (int col = 0; col < 64; col++) {
                alphabet[row][col] = chars.charAt((row + col) % 64);
            }
        }
    }

    // Metodos
    public String encrypt(String message) {
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < message.length(); i++) {
            char current = message.charAt(i);
            if (current == ' ') {
                result.append(' ');
                continue;
            }
            int row = chars.indexOf(current);
            int col = key[i % key.length];
            if (row == -1 || col == -1) {
                result.append(current);
            } else {
                result.append(alphabet[row][col]);
            }
        }
        return result.toString();
    }
    public String decrypt(String encrypted) {
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < encrypted.length(); i++) {
            char current = encrypted.charAt(i);
            if (current == ' ') {
                result.append(' ');
                continue;
            }
            int col = key[i%key.length];
            int row = -1;
            for (int r=0; r<64; r++) {
                if (alphabet[r][col]==current) {
                    row=r;
                    break;
                }
            }
            if (row!=-1) {
                result.append(chars.charAt(row));
            } else {
                result.append(current);
            }
        }
        return result.toString();
    }
    public void reEncrypt() {
        System.out.print("Frase a encriptar: ");
        String encrypted = scanner.nextLine();
        String original = decrypt(encrypted);
        System.out.print("Nueva Key: ");
        String newKey = scanner.nextLine();
        setKey(newKey);
        String newEncrypted = encrypt(original);
        System.out.println("Frase reencriptada: " + newEncrypted);
    }
    public char search(int position) {
        for (int i = 0; i < alphabet.length; i++) {
            for (int j = 0; j < alphabet.length; j++) {
                if (position == i * alphabet.length + j) {
                    return alphabet[i][j];
                }
            }
        }
        return '_';
    }
    public char optimalSearch(int position) {
        if (position < 0 || position >= 64 * 64) return '_';
        return alphabet[position / 64][position % 64];
    }

}
